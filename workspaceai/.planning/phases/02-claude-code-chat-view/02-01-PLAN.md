    After the renderer.html override, add the renderer.code override BEFORE the marked.use call:
      renderer.code = ({ text, lang }: Tokens.Code): string => {
        if (lang && hljs.getLanguage(lang)) {
          const highlighted = hljs.highlight(text, { language: lang }).value;
          return '<pre><code class="hljs language-' + lang + '">' + highlighted + '</code></pre>';
        }
        const escaped = text
          .replace(/&/g, '&amp;')
          .replace(/</g, '&lt;')
          .replace(/>/g, '&gt;');
        return '<pre><code>' + escaped + '</code></pre>';
      };

    CRITICAL: renderer.code must come before the marked.use({ renderer, ... }) call.
    CRITICAL: Do NOT escape the highlighted output — hljs.highlight().value is already
